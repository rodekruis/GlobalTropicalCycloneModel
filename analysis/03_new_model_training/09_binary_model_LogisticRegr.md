```python
# Define a threshold to separate target into damaged and not_damaged
thres = 10.0

        df.at[i, "binary_damage"] = 1
    else:
        df.at[i, "binary_damage"] = 0

df["binary_damage"] = df["binary_damage"].astype("int")
```
